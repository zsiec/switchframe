> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a unified slide-out I/O management panel that replaces the SRT output modal, showing all inputs (sources) and outputs (destinations) with CRUD operations, stats, and per-destination SCTE-35 toggle.

```typescript
export function createSRTSource(config: CreateSRTSourceConfig): Promise<{ key: string }> {
	return request('/api/sources', {
		method: 'POST',
		headers: { 'Content-Type': 'application/json' },
		body: JSON.stringify(config),
	});
}

export function deleteSRTSource(key: string): Promise<void> {
	return request(`/api/sources/${encodeURIComponent(key)}`, {
		method: 'DELETE',
	});
}

export function getSRTSourceStats(key: string): Promise<SRTSourceStats> {
	return request(`/api/sources/${encodeURIComponent(key)}/srt/stats`);
}

export function updateSRTLatency(key: string, latencyMs: number): Promise<void> {
	return request(`/api/sources/${encodeURIComponent(key)}/srt`, {
		method: 'PUT',
		headers: { 'Content-Type': 'application/json' },
		body: JSON.stringify({ latencyMs }),
	});
}
```

```svelte
<script lang="ts">
	import type { ControlRoomState } from '$lib/api/types';
	import { getConfirmMode, setConfirmMode } from '$lib/state/preferences.svelte';
	import Clock from './Clock.svelte';
	import RecordingControl from './RecordingControl.svelte';
	import ConnectionStatus from './ConnectionStatus.svelte';

	type ConnectionIndicatorState = 'webtransport' | 'polling' | 'disconnected';

	interface Props {
		state: ControlRoomState;
		connectionState?: ConnectionIndicatorState;
		switchLayout?: () => void;
		onToggleIOPanel?: () => void;
		ioPanelVisible?: boolean;
	}
	let { state: crState, connectionState = 'disconnected', switchLayout, onToggleIOPanel, ioPanelVisible = false }: Props = $props();

	let thumbKey = $state(0);
	let thumbInterval: ReturnType<typeof setInterval> | undefined;

	const srtActive = $derived(crState.srtOutput?.active ?? false);
	const recActive = $derived(crState.recording?.active ?? false);
	const showConfidence = $derived(recActive || srtActive);
	const hasDestinations = $derived((crState.destinations?.length ?? 0) > 0);
	const hasIOActivity = $derived(srtActive || hasDestinations);

	// Check for any unhealthy SRT inputs or errored outputs
	const hasIOWarning = $derived.by(() => {
		const sources = Object.values(crState.sources);
		const unhealthyInput = sources.some(s => s.type === 'srt' && s.status !== 'healthy');
		const erroredOutput = crState.destinations?.some(d => d.state === 'error') ?? false;
		return unhealthyInput || erroredOutput;
	});

	$effect(() => {
		if (showConfidence) {
			thumbInterval = setInterval(() => { thumbKey++; }, 1000);
		} else {
			if (thumbInterval) clearInterval(thumbInterval);
			thumbInterval = undefined;
		}
		return () => { if (thumbInterval) clearInterval(thumbInterval); };
	});
</script>

<div class="output-controls">
	<Clock />
	<ConnectionStatus state={connectionState} />
	{#if showConfidence}
		<img class="confidence-thumb" src="/api/output/confidence?t={thumbKey}" alt="Program output" width="80" height="45" />
	{/if}
	<RecordingControl state={crState} />
	<button
		class="header-btn io-btn"
		class:io-active={ioPanelVisible || hasIOActivity}
		class:io-warning={hasIOWarning}
		onclick={() => onToggleIOPanel?.()}
		title="I/O Management (Shift+I)"
	>I/O</button>

		title="Require double-press of Space or Shift+number for hot punches"
	>CONFIRM</button>
	{#if switchLayout}
		<button class="header-btn mode-btn" onclick={switchLayout} title="Switch layout mode">MODE</button>

</div>
```
